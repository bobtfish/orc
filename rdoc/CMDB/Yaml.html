def save_application(spec, groups)
  filename = convention(spec)
  dir = Pathname.new(filename).dirname
  if !dir.exist?
      Dir.mkdir dir
  end
  File.open( filename, "w" ) do |f|
    flattened = groups.map { |el| Hash[el.map{|(k,v)| [k.to_s,v]}] }
    f.write( flattened.to_yaml )
  end
end
